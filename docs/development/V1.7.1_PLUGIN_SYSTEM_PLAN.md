```tsx
// frontend/src/pages/admin/PluginManager.tsx

interface PluginCardProps {
  id: string;
  name: string;
  description: string;
  installed: boolean;
  enabled: boolean;
  version: string;
  stats?: { total_items?: number; total_sessions?: number };
}

const PluginCard: React.FC<PluginCardProps> = ({ ...props }) => {
  const [enabled, setEnabled] = useState(props.enabled);

  const handleToggle = async () => {
    await adminPluginsApi.togglePlugin(props.id, !enabled);
    setEnabled(!enabled);
  };

  return (
    <div className="plugin-card">
      <div className="flex justify-between items-center">
        <div>
          <h3>{props.name}</h3>
          <p>{props.description}</p>
          {props.stats && (
            <div className="stats">
              {props.stats.total_items && <span>{props.stats.total_items} tutorials</span>}
              {props.stats.total_sessions && <span>{props.stats.total_sessions} game sessions</span>}
            </div>
          )}
        </div>

        <div className="actions">
          {props.installed ? (
            <>
              <Toggle checked={enabled} onChange={handleToggle} />
              <Button variant="outline">Configure</Button>
            </>
          ) : (
            <Button>Install</Button>
          )}
        </div>
      </div>
    </div>
  );
};

export const PluginManager: React.FC = () => {
  const [plugins, setPlugins] = useState<PluginStatus[]>([]);

  useEffect(() => {
    adminPluginsApi.getPluginsStatus().then(setPlugins);
  }, []);

  return (
    <div className="plugin-manager">
      <h1>Plugin Manager</h1>
      <p>Enable or disable features to customize your platform</p>

      <div className="plugins-grid">
        {Object.entries(plugins).map(([id, plugin]) => (
          <PluginCard key={id} id={id} {...plugin} />
        ))}
      </div>
    </div>
  );
};
```
